    获取单词&正则匹配

    获取单词

    import re
# 单词写入文件
file = open("./EngWords.txt", 'w') 
lines = None
wordscount = 0
# 正则匹配 `［`之前的单词
compiled_pattern = re.compile((r'\w+(?=［)'))
with open('./四级词汇词根+联想记忆法-俞敏洪.txt') as f:
    lines = f.readlines()
for line in lines:
    #sub = re.search(r'\w+(?=［)', line)
    sub = compiled_pattern.search(line)
    if sub is not None:
        wordscount +=1
        file.write(sub.group(0) + '\n')
file.close() 
print(wordscount)
